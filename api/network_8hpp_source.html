89 {
90 // Continuously check if the received can message is valid. NOTE: that, in
91 // general, the looping logic for this function almost never occurs as
92 // copies from RAM/cache are fast between receive_handler and GetMessage()
93 while (true) {
94 // First, atomically copy the access_counter to a variable.
95 const auto read_message_start = m_access_counter.load();
96
97 // Copy the contents of the message into the can_message variable.
98 const can::message_t can_message = m_data;
99
100 // Finally, atomically copy the access_counter again such that the value
101 // between the start and finish can be compared.
102 const auto read_message_finish = m_access_counter.load();
103
104 // If and only if these two values are the same, meaning that the
105 // receive_handler isn't currently modifying the node_t's data,
106 // can the message be considered valid.
107 if (read_message_start == read_message_finish) {
108 return can_message;
109 }
110 }

127 {
128 // Atomic increment of the access counter to notify any threads that are
129 // using GetMessage() that the value of this node is changing.

240 void receive_handler(const can::message_t& p_message) noexcept
241 {
242 // Check if the map already has a value for this ID. This acts as the last
243 // stage of the can filter for the can bus network module. If the key
244 // does NOT exist in the map, then this message will not be saved.
245 // Typically, this only happens when the hardware filter has not been setup
246 // properly to eliminate can messages that should be ignored.
247 //
248 // Map lookups can be costly, especially in a interrupt context, so only
249 // needing to hash/lookup the ID once is preferred. To prevent multiple
250 // lookups, an iterator is stored into the message_node variable.
251 auto message_node = m_messages.find(p_message.id);
252
253 // If the ID has an associated value in the map, then the node std::pair<>
254 // is returned, otherwise, the ::end() node is returned.
255 if (message_node != m_messages.end()) {
256 message_node->second.update(p_message);
257 }
