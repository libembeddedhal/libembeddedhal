91 // general, the looping logic for this function almost never occurs as
92 // copies from RAM/cache are fast between receive_handler and GetMessage()
93 while (true) {

101 // between the start and finish can be compared.

105 // receive_handler isn't currently modifying the node_t's data,
106 // can the message be considered valid.

239 void receive_handler(const can::message_t& p_message) noexcept
240 {
241 // Check if the map already has a value for this ID. This acts as the last
242 // stage of the can filter for the can bus network module. If the key
243 // does NOT exist in the map, then this message will not be saved.
244 // Typically, this only happens when the hardware filter has not been setup
245 // properly to eliminate can messages that should be ignored.
246 //
247 // Map lookups can be costly, especially in a interrupt context, so only
248 // needing to hash/lookup the ID once is preferred. To prevent multiple
249 // lookups, an iterator is stored into the message_node variable.
250 auto message_node = m_messages.find(p_message.id);
251
252 // If the ID has an associated value in the map, then the node std::pair<>
253 // is returned, otherwise, the ::end() node is returned.
254 if (message_node != m_messages.end()) {
255 message_node->second.update(p_message);
