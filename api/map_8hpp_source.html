77template<typename T>
78[[nodiscard]] constexpr boost::leaf::result<T> map(
79 T p_target,
80 map_range<T> p_input_range,
81 map_range<T> p_output_range) noexcept
82{
83 static_assert(sizeof(T) <= sizeof(std::uint32_t));
84
85 // Use x & y directly here because distance will do a range check itself
86 auto output_distance = p_output_range.distance();
87 auto input_distance = p_input_range.distance();
88
89 // If the output range's distance is 0 it means that x & y are both the same
90 // value. Thus all target values map to the same value in the output and thus
91 // the result is simply x or y.
92 if (std::cmp_equal(output_distance, 0)) {
93 return p_output_range.x;
94 }
95 // If the input range is 0, then the input range is a single number. Mapping a
96 // 1-dimensional range to a 2-D range is not defined and thus the inputs are
97 // not valid.
98 //
99 // Throw error std::errc::invalid_argument.
100 if (std::cmp_equal(input_distance, 0)) {
101 return boost::leaf::new_error(std::errc::invalid_argument);
102 }
103
104 // Bounds check p_target and if it is is outside of the input range then
105 // return clamped the return value to the bounds of the output range.
106 if (p_target >= p_input_range.max()) {
107 return p_output_range.max();
108 } else if (p_target <= p_input_range.min()) {
109 return p_output_range.min();
110 }
111
112 // Verify at compile time that the types for "output_distance" &
113 // "input_distance" are both unsigned 32 bit integers
114 static_assert(std::is_same_v<decltype(output_distance), std::uint32_t>);
115 static_assert(std::is_same_v<decltype(input_distance), std::uint32_t>);
116
117 // Performing integer division & modulus which will floor the result and
118 // provide the whole number solution. This operation is safe from "divide by
119 // zero" errors as because input_distance is checked for being zero above.
120 auto slope_quotient = output_distance / input_distance;
121 auto slope_remainder = output_distance % input_distance;
122
123 // Operation is well defined for integers 32-bit and below
124 auto decimal = embed::percent::from_ratio(slope_remainder, input_distance);
125
126 // p_target has already been bounds checked and shown to be within the bounds
127 // of p_input_range, thus this subtracts operation will never yield an
128 // underflow or overflow.
129 auto align_target_to_zero = p_target - p_input_range.min();
130
131 // This operation will never overflow because the ratio between two ranges
132 // confined to the integral type T times the zero aligned target value will
133 // never exceed the bounds of T.
134 auto quotient_scaled_value = slope_quotient * align_target_to_zero;
135 // Multiplication between embed::percent & a 32-bit number is always a safe
136 // operation.
137 auto remainder_scaled_value = decimal * align_target_to_zero;
138
139 // Summing the two values will produce a value within the bounds of the output
140 // range's distance.
141 auto scaled_value = quotient_scaled_value + remainder_scaled_value;
142 // Adding the output range's minimum value will push the scaled value directly
143 // to where it should be mapped in the output range.
144 auto scaled_value_with_offset = p_output_range.min() + scaled_value;
145
146 return scaled_value_with_offset;
147}
