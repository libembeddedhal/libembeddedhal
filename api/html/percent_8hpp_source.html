57template<std::integral T, size_t SourceWidth, std::integral U>
58[[nodiscard]] constexpr T upscale_integer(U p_value) noexcept
59{
60 constexpr size_t output_bit_width = sizeof(T) * CHAR_BIT;
61
62 static_assert(output_bit_width >= SourceWidth,
63 "The destination bit width must be equal to or greater than "
64 "the type of the value to be scaled up.");
65
66 static_assert(SourceWidth >= 2, "Bit Width must be greater than ");
67
68 constexpr size_t final_source_width =
69 (std::is_signed_v<U>) ? SourceWidth - 1 : SourceWidth;
70 constexpr size_t final_width =
71 (std::is_signed_v<T>) ? output_bit_width - 1 : output_bit_width;
72
73 // Calculate the difference between the destination and the source width
74 constexpr size_t width_difference = final_width - final_source_width;
75
76 T result = static_cast<T>(p_value << width_difference);
77
78 if (p_value > 0) {
79 for (size_t i = final_source_width; i < final_width;
80 i += final_source_width) {
81 result |= (result >> i);
82 }
83 }
84
85 return result;
86}

318 [[nodiscard]] static constexpr percent from_ratio(T p_progress,
319 T p_maximum) noexcept
320 {
321 static_assert(sizeof(T) <= sizeof(std::int32_t),
322 "T must not exceed 32 bits in width.");
323 overflow_t result = p_progress;
324 // Multiplication between result, which must be size 32-bits or lower times
325 // raw_max() which is defined as 32-bit can be held in the overflow type
326 // which is int64_t, thus no overflow can occur here.
327 result = (result * raw_max());
328 // If p_progress is less than p_maximum then the result value will fit
329 // within a 32-bit integer.
330 result = rounding_division(result, overflow_t(p_maximum));
331 // If p_progress was greater than p_maximum, then the end result will be
332 // greater than raw_max() or less than raw_min().
333 // This stage will clamp the result to between the 32-bit.
334 result = std::clamp(result, raw_min(), raw_max());
335
336 return percent(static_cast<int_t>(result));
337 }

446 [[nodiscard]] auto to_string() const noexcept
447 {
448 constexpr int_t fixed_percent_scalar = 1000000000;
449 // Based on the number of characters needed to hold "fixed_percent_scalar"
450 // as well as a '.' and a '-' sign
451 constexpr size_t scalar_length = 12;
452
453 if (raw_value() >= raw_max() - 2) {
454 return to_array<scalar_length>("+1.000000000");
455 } else if (raw_value() <= raw_min() + 2) {
456 return to_array<scalar_length>("-1.000000000");
457 }
458
459 // Make a copy of the percent value
460 percent absolute_percent = *this;
461 // Check and save if the number is negative
462 bool negative = absolute_percent.raw_value() < 0;
463 // If negative absolute, make the percent positive
464 if (negative) {
465 absolute_percent.m_value = absolute_percent.raw_value() * -1;
466 }
467
468 // +1 for a '\0' at the end
469 std::array<char, scalar_length + 1> buffer{ '\0' };
470 // +2 for a '.' and a '\0' character at the end
471 std::array<char, scalar_length + 1> percent_string{ '\0' };
472
473 // Scale "fixed_percent_scalar" by the absolute_percent value
474 int_t decimal_percent = absolute_percent * fixed_percent_scalar;
475
476 // Convert to characters
477 auto char_conversion_result =
478 std::to_chars(buffer.begin(), buffer.end(), decimal_percent);
479
480 size_t string_length = char_conversion_result.ptr - buffer.data();
481 size_t leading_zeros = scalar_length - string_length;
482
483 // Add the leading +/- sign
484 percent_string[0] = (negative) ? '-' : '+';
485
486 // Add the '0' and '.'. We know that the value must be below 1.0 and above
487 // -1.0 because if the values were 1.0 or -1.0 then we would have returned
488 // them at the start of the function.
489 percent_string[1] = '0';
490 percent_string[2] = '.';
491
492 // Add any leading zeros into the percent string array
493 for (size_t i = 3; i < leading_zeros; i++) {
494 percent_string[i] = '0';
495 }
496 // Now copy the contents of the character buffer into the percent string
497 // offset by the number of leading zeros.
498 std::copy_n(
499 buffer.begin(), string_length, percent_string.begin() + leading_zeros);
500
501 return percent_string;
502 }
