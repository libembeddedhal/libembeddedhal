51 [[nodiscard]] explicit constexpr operator percent() const noexcept
52 {
53 std::uint32_t total_cycles = 0;
54 std::uint32_t scaled_high = 0;
55 // Convert high & low to int64_t to prevent overflow
56 std::uint64_t overflow_total = std::uint64_t{ high } + std::uint64_t{ low };
57
58 // If overflow_total is greater than what can fit within an uint32_t, then
59 // we can shift the values over by 1 to get them to fit. This effectively
60 // divides the two values in half, but the ratio will stay the same.
61 if (overflow_total > std::numeric_limits<decltype(high)>::max()) {
62 total_cycles = static_cast<std::uint32_t>(overflow_total >> 1);
63 scaled_high = high >> 1;
64 } else {
65 total_cycles = static_cast<std::uint32_t>(overflow_total);
66 scaled_high = high;
67 }
68
69 return percent::from_ratio(scaled_high, total_cycles);
70 }

150 [[nodiscard]] std::int64_t cycles_per(
151 embed::time_duration p_duration) const noexcept
152 {
153 // Full Equation:
154 // =========================================================================
155 //
156 // / ratio_num \_
157 // frequency_hz * |period| * | ----------- | = cycles
158 // \ ratio_den /
159 //
160 // std::chrono::nanoseconds::period::num == 1
161 // std::chrono::nanoseconds::period::den == 1,000,000
162
163 constexpr std::int64_t numerator = decltype(p_duration)::period::num;
164 constexpr std::int64_t denominator = decltype(p_duration)::period::den;
165 // Storing 32-bit value in a std::uint64_t for later computation, no
166 // truncation possible.
167 const std::int64_t duration = absolute_value(p_duration.count());
168 // Duration contains at most a 64-bit number, cycles_per_second() is a
169 // 32-bit number, and numerator is always the value 1.
170 //
171 // To contain the maximum resultant possible requires storage within an
172 // integer of size 64-bits, and thus will fit within a std::uint64_t
173 // variable, no overflow checks required.
174 const std::int64_t count = duration * cycles_per_second() * numerator;
175 const std::int64_t cycle_count = rounding_division(count, denominator);
176
177 return cycle_count;
178 }

188 template<typename Period = std::femto>
189 std::chrono::duration<int64_t, Period> wavelength() const
190 {
191 // Full Equation (based on the equation in cycles_per()):
192 // =========================================================================
193 //
194 // / cycle_count * ratio_den \_
195 // |period| = | ---------------------------|
196 // \ ratio_num * frequency_hz /
197 //
198 // let cycle_count = 1
199 // --> Wavelength is the length of a single cycle
200 //
201 // let ratio_num = 1
202 // --> Smallest frequency can only be as small as 1Hz. No wavelengths with
203 // numerators
204
205 static_assert(Period::num == 1,
206 "Period::num of 1 are allowed for this function.");
207 static_assert(Period::den <= 1000000000000000000,
208 "Period::den cannot exceed 1000000000000000000.");
209
210 std::uint64_t numerator = Period::den;
211 std::uint64_t denominator = cycles_per_second();
212 std::uint64_t duration = rounding_division(numerator, denominator);
213 return std::chrono::duration<int64_t, Period>(duration);
214 }

224 [[nodiscard]] std::chrono::nanoseconds duration_from_cycles(
225 std::int32_t p_cycles) const noexcept
226 {
227 // Full Equation (based on the equation in cycles_per()):
228 // =========================================================================
229 //
230 // / cycles * ratio_den \_
231 // |period| = | ---------------------------|
232 // \ frequency_hz * ratio_num /
233 //
234 constexpr auto time_duration_den = std::chrono::nanoseconds::period::den;
235 constexpr auto time_duration_num = std::chrono::nanoseconds::period::num;
236 constexpr auto scale_den = std::int64_t{ time_duration_den };
237 constexpr auto scale_num = std::int64_t{ time_duration_num };
238
239 std::int64_t numerator = std::int64_t{ p_cycles } * scale_den;
240 std::int64_t denominator = cycles_per_second() * scale_num;
241 std::int64_t nanoseconds = rounding_division(numerator, denominator);
242
243 return std::chrono::nanoseconds(nanoseconds);
244 }
