52 [[nodiscard]] explicit constexpr operator percent() const noexcept
53 {
54 std::uint32_t total_cycles = 0;
55 std::uint32_t scaled_high = 0;
56 // Convert high & low to int64_t to prevent overflow
57 std::uint64_t overflow_total = std::uint64_t{ high } + std::uint64_t{ low };
58
59 // If overflow_total is greater than what can fit within an uint32_t, then
60 // we can shift the values over by 1 to get them to fit. This effectively
61 // divides the two values in half, but the ratio will stay the same.
62 if (overflow_total > std::numeric_limits<decltype(high)>::max()) {
63 total_cycles = static_cast<std::uint32_t>(overflow_total >> 1);
64 scaled_high = high >> 1;
65 } else {
66 total_cycles = static_cast<std::uint32_t>(overflow_total);
67 scaled_high = high;
68 }
69
70 return percent::from_ratio(scaled_high, total_cycles);
71 }

163 [[nodiscard]] std::int64_t cycles_per(
164 embed::time_duration p_duration) const noexcept
165 {
166 // Full Equation:
167 // =========================================================================
168 //
169 // / ratio_num \_
170 // frequency_hz * |period| * | ----------- | = cycles
171 // \ ratio_den /
172 //
173 // std::chrono::nanoseconds::period::num == 1
174 // std::chrono::nanoseconds::period::den == 1,000,000
175
176 constexpr std::int64_t numerator = decltype(p_duration)::period::num;
177 constexpr std::int64_t denominator = decltype(p_duration)::period::den;
178 // Storing 32-bit value in a std::uint64_t for later computation, no
179 // truncation possible.
180 const std::int64_t duration = absolute_value(p_duration.count());
181 // Duration contains at most a 64-bit number, cycles_per_second() is a
182 // 32-bit number, and numerator is always the value 1.
183 //
184 // To contain the maximum resultant possible requires storage within an
185 // integer of size 64-bits, and thus will fit within a std::uint64_t
186 // variable, no overflow checks required.
187 const std::int64_t count = duration * cycles_per_second() * numerator;
188 const std::int64_t cycle_count = rounding_division(count, denominator);
189
190 return cycle_count;
191 }

201 template<typename Period = std::femto>
202 std::chrono::duration<int64_t, Period> wavelength() const
203 {
204 // Full Equation (based on the equation in cycles_per()):
205 // =========================================================================
206 //
207 // / cycle_count * ratio_den \_
208 // |period| = | ---------------------------|
209 // \ ratio_num * frequency_hz /
210 //
211 // let cycle_count = 1
212 // --> Wavelength is the length of a single cycle
213 //
214 // let ratio_num = 1
215 // --> Smallest frequency can only be as small as 1Hz. No wavelengths with
216 // numerators
217
218 static_assert(Period::num == 1,
219 "Period::num of 1 are allowed for this function.");
220 static_assert(Period::den <= 1000000000000000000,
221 "Period::den cannot exceed 1000000000000000000.");
222
223 std::uint64_t numerator = Period::den;
224 std::uint64_t denominator = cycles_per_second();
225 std::uint64_t duration = rounding_division(numerator, denominator);
226 return std::chrono::duration<int64_t, Period>(duration);
227 }

237 [[nodiscard]] std::chrono::nanoseconds duration_from_cycles(
238 std::int32_t p_cycles) const noexcept
239 {
240 // Full Equation (based on the equation in cycles_per()):
241 // =========================================================================
242 //
243 // / cycles * ratio_den \_
244 // |period| = | ---------------------------|
245 // \ frequency_hz * ratio_num /
246 //
247 constexpr auto time_duration_den = std::chrono::nanoseconds::period::den;
248 constexpr auto time_duration_num = std::chrono::nanoseconds::period::num;
249 constexpr auto scale_den = std::int64_t{ time_duration_den };
250 constexpr auto scale_num = std::int64_t{ time_duration_num };
251
252 std::int64_t numerator = std::int64_t{ p_cycles } * scale_den;
253 std::int64_t denominator = cycles_per_second() * scale_num;
254 std::int64_t nanoseconds = rounding_division(numerator, denominator);
255
256 return std::chrono::nanoseconds(nanoseconds);

313 template<typename InputIt>
314 [[nodiscard]] constexpr InputIt closest(InputIt p_first,
315 InputIt p_last,
316 frequency p_target,
317 selection_mode p_selection_mode)
318 {
319 auto cost = [p_target](frequency& p_candidate) -> std::uint32_t {
320 return distance(p_candidate.cycles_per_second(),
321 p_target.cycles_per_second());
322 };
323
324 auto is_applicable = [p_target,
325 p_selection_mode](frequency& p_candidate) -> bool {
326 switch (p_selection_mode) {
327 case selection_mode::lower:
328 return p_candidate <= p_target;
329 case selection_mode::higher:
330 return p_candidate >= p_target;
331 case selection_mode::closest:
332 return true;
333 default:
334 return false;
335 }
336 };
337
338 InputIt best = p_last;
339 auto best_cost = std::numeric_limits<uint32_t>::max();
340
341 for (InputIt it = p_first; it != p_last; it++) {
342 frequency candidate{ *this / *it };
343 if (is_applicable(candidate) && cost(candidate) < best_cost) {
344 best = it;
345 best_cost = cost(candidate);
346 }
347 }
348 return best;
349 }
