51 [[nodiscard]] explicit constexpr operator percent() const noexcept
52 {
53 std::uint32_t total_cycles = 0;
54 std::uint32_t scaled_high = 0;
55 // Convert high & low to int64_t to prevent overflow
56 std::uint64_t overflow_total = std::uint64_t{ high } + std::uint64_t{ low };
57
58 // If overflow_total is greater than what can fit within an uint32_t, then
59 // we can shift the values over by 1 to get them to fit. This effectively
60 // divides the two values in half, but the ratio will stay the same.

62 total_cycles = static_cast<std::uint32_t>(overflow_total >> 1);

64 } else {
65 total_cycles = static_cast<std::uint32_t>(overflow_total);
66 scaled_high = high;
67 }

69 return percent::from_ratio(scaled_high, total_cycles);
70 }

171[[nodiscard]] constexpr std::int64_t cycles_per(
172 const frequency& p_source,
173 embed::time_duration p_duration) noexcept
174{
175 // Full Equation:
176 // =========================================================================
177 //
178 // / ratio_num \_
179 // frequency_hz * |period| * | ----------- | = cycles
180 // \ ratio_den /
181 //
182 // std::chrono::nanoseconds::period::num == 1
183 // std::chrono::nanoseconds::period::den == 1,000,000
184
185 constexpr std::int64_t numerator = decltype(p_duration)::period::num;
186 constexpr std::int64_t denominator = decltype(p_duration)::period::den;
187 // Storing 32-bit value in a std::uint64_t for later computation, no
188 // truncation possible.
189 const std::int64_t duration = absolute_value(p_duration.count());
190 // Duration contains at most a 64-bit number, value() is a
191 // 32-bit number, and numerator is always the value 1.
192 //
193 // To contain the maximum resultant possible requires storage within an
194 // integer of size 64-bits, and thus will fit within a std::uint64_t
195 // variable, no overflow checks required.
196 const std::int64_t count = duration * p_source.value_hz * numerator;
197 const std::int64_t cycle_count = rounding_division(count, denominator);
198
199 return cycle_count;
200}

210template<typename Period = std::femto>
211constexpr std::chrono::duration<int64_t, Period> wavelength(
212 const frequency& p_source)
213{
214 // Full Equation (based on the equation in cycles_per()):
215 // =========================================================================
216 //
217 // / cycle_count * ratio_den \_
218 // |period| = | ---------------------------|
219 // \ ratio_num * frequency_hz /
220 //
221 // let cycle_count = 1
222 // --> Wavelength is the length of a single cycle
223 //
224 // let ratio_num = 1
225 // --> Smallest frequency can only be as small as 1Hz. No wavelengths with
226 // numerators
227
228 static_assert(Period::num == 1,
229 "Period::num of 1 are allowed for this function.");
230 static_assert(Period::den <= 1000000000000000000,
231 "Period::den cannot exceed 1000000000000000000.");
232
233 std::uint64_t numerator = Period::den;
234 std::uint64_t denominator = p_source.value_hz;
235 std::uint64_t duration = rounding_division(numerator, denominator);
236 return std::chrono::duration<int64_t, Period>(duration);
237}

248[[nodiscard]] constexpr std::chrono::nanoseconds duration_from_cycles(
249 const frequency& p_source,
250 std::int32_t p_cycles) noexcept
251{
252 // Full Equation (based on the equation in cycles_per()):
253 // =========================================================================
254 //
255 // / cycles * ratio_den \_
256 // |period| = | ---------------------------|
257 // \ frequency_hz * ratio_num /
258 //
259 constexpr auto time_duration_den = std::chrono::nanoseconds::period::den;
260 constexpr auto time_duration_num = std::chrono::nanoseconds::period::num;
261 constexpr auto scale_den = std::int64_t{ time_duration_den };
262 constexpr auto scale_num = std::int64_t{ time_duration_num };
263
264 std::int64_t numerator = std::int64_t{ p_cycles } * scale_den;
265 std::int64_t denominator = p_source.value_hz * scale_num;
266 std::int64_t nanoseconds = rounding_division(numerator, denominator);
267
268 return std::chrono::nanoseconds(nanoseconds);
269}

323template<typename InputIt>
324[[nodiscard]] constexpr InputIt closest(const frequency& p_source,
325 InputIt p_first,
326 InputIt p_last,
327 frequency p_target,
328 divider_rule p_divider_rule)
329{
330 auto cost = [p_target](frequency& p_candidate) -> std::uint32_t {
331 return distance(p_candidate.value_hz, p_target.value_hz);
332 };
333
334 auto is_applicable = [p_target,
335 p_divider_rule](frequency& p_candidate) -> bool {
336 switch (p_divider_rule) {
337 case divider_rule::lower:
338 return p_candidate <= p_target;
339 case divider_rule::higher:
340 return p_candidate >= p_target;
341 case divider_rule::closest:
342 return true;
343 default:
344 return false;
345 }
346 };
347
348 InputIt best = p_last;
349 auto best_cost = std::numeric_limits<uint32_t>::max();
350
351 for (InputIt it = p_first; it != p_last; it++) {
352 frequency candidate{ p_source / *it };
353 if (is_applicable(candidate) && cost(candidate) < best_cost) {
354 best = it;
355 best_cost = cost(candidate);
356 }
357 }
358 return best;
359}
