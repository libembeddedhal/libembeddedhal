62template<std::integral T, size_t SourceWidth, std::integral U>
63[[nodiscard]] constexpr T upscale_integer(U p_value) noexcept
64{
65 constexpr size_t output_bit_width = sizeof(T) * CHAR_BIT;
66
67 static_assert(output_bit_width >= SourceWidth,
68 "The destination bit width must be equal to or greater than "
69 "the type of the value to be scaled up.");
70
71 static_assert(SourceWidth >= 2, "Bit Width must be greater than ");
72
73 constexpr size_t final_source_width =
74 (std::is_signed_v<U>) ? SourceWidth - 1 : SourceWidth;
75 constexpr size_t final_width =
76 (std::is_signed_v<T>) ? output_bit_width - 1 : output_bit_width;
77
78 // Calculate the difference between the destination and the source width
79 constexpr size_t width_difference = final_width - final_source_width;
80
81 T result = static_cast<T>(p_value << width_difference);
82
83 if (p_value > 0) {
84 for (size_t i = final_source_width; i < final_width;
85 i += final_source_width) {
86 result |= (result >> i);
87 }
88 }
89
90 return result;
91}

225 {
226 static_assert(sizeof(T) <= sizeof(std::uint32_t),
227 "T must not exceed 32 bits in width.");
228 overflow_t result = p_progress;
229 // Multiplication between result, which must be size 32-bits or lower times
230 // raw_max() which is defined as 32-bit can be held in the overflow type
231 // which is int64_t, thus no overflow can occur here.
232 result = (result * raw_max());
233 // If p_progress is less than p_maximum then the result value will fit
234 // within a 32-bit integer.
235 result = rounding_division(result, overflow_t(p_maximum));
236 // If p_progress was greater than p_maximum, then the end result will be
237 // greater than raw_max() or less than raw_min().
238 // This stage will clamp the result to between the 32-bit.
239 result = std::clamp(result, raw_min(), raw_max());

529[[nodiscard]] constexpr auto to_string(const percent& p_percent) noexcept
530{
531 using int_t = percent::int_t;
532
533 constexpr int_t fixed_percent_scalar = 1000000000;
534 // Based on the number of characters needed to hold "fixed_percent_scalar"
535 // as well as a '.' and a '-' sign
536 constexpr size_t scalar_length = 12;
537
538 if (p_percent == percent::max()) {
539 return to_array<scalar_length>("+1.000000000");
540 } else if (p_percent == percent::min()) {
541 return to_array<scalar_length>("-1.000000000");
542 }
543
544 percent positive_percent = p_percent;
545 // If negative absolute, make the percent positive
546 bool negative = p_percent.raw_value() < 0;
547 if (negative) {
548 positive_percent = -p_percent;
549 }
550
551 // +1 for a '\0' at the end
552 std::array<char, scalar_length + 1> buffer{ '\0' };
553 // +2 for a '.' and a '\0' character at the end
554 std::array<char, scalar_length + 1> percent_string{ '\0' };
555
556 // Scale "fixed_percent_scalar" by the absolute_percent value
557 int_t decimal_percent = positive_percent * fixed_percent_scalar;
558
559 // Convert to characters
560 auto char_conversion_result =
561 std::to_chars(buffer.begin(), buffer.end(), decimal_percent);
562
563 size_t string_length = char_conversion_result.ptr - buffer.data();
564 size_t leading_zeros = scalar_length - string_length;
565
566 // Add the leading +/- sign
567 percent_string[0] = (negative) ? '-' : '+';
568
569 // Add the '0' and '.'. We know that the value must be below 1.0 and above
570 // -1.0 because if the values were 1.0 or -1.0 then we would have returned
571 // them at the start of the function.
572 percent_string[1] = '0';
573 percent_string[2] = '.';
574
575 // Add any leading zeros into the percent string array
576 for (size_t i = 3; i < leading_zeros; i++) {
577 percent_string[i] = '0';
578 }
579 // Now copy the contents of the character buffer into the percent string
580 // offset by the number of leading zeros.
581 std::copy_n(
582 buffer.begin(), string_length, percent_string.begin() + leading_zeros);
583
584 return percent_string;
585}
