63[[nodiscard]] constexpr T upscale_integer(U p_value) noexcept
64{
65 constexpr size_t output_bit_width = sizeof(T) * CHAR_BIT;
66
67 static_assert(output_bit_width >= SourceWidth,
68 "The destination bit width must be equal to or greater than "
69 "the type of the value to be scaled up.");
70
71 static_assert(SourceWidth >= 2, "Bit Width must be greater than ");
72
73 constexpr size_t final_source_width =
74 (std::is_signed_v<U>) ? SourceWidth - 1 : SourceWidth;
75 constexpr size_t final_width =
76 (std::is_signed_v<T>) ? output_bit_width - 1 : output_bit_width;
77
78 // Calculate the difference between the destination and the source width
79 constexpr size_t width_difference = final_width - final_source_width;
80
81 T result = static_cast<T>(p_value << width_difference);
82
83 if (p_value > 0) {
84 for (size_t i = final_source_width; i < final_width;
85 i += final_source_width) {
86 result |= (result >> i);
87 }
88 }
89
90 return result;
91}

323 [[nodiscard]] static constexpr percent from_ratio(T p_progress,
324 T p_maximum) noexcept
325 {
326 static_assert(sizeof(T) <= sizeof(std::int32_t),
327 "T must not exceed 32 bits in width.");
328 overflow_t result = p_progress;
329 // Multiplication between result, which must be size 32-bits or lower times
330 // raw_max() which is defined as 32-bit can be held in the overflow type
331 // which is int64_t, thus no overflow can occur here.
332 result = (result * raw_max());
333 // If p_progress is less than p_maximum then the result value will fit
334 // within a 32-bit integer.
335 result = rounding_division(result, overflow_t(p_maximum));
336 // If p_progress was greater than p_maximum, then the end result will be
337 // greater than raw_max() or less than raw_min().
338 // This stage will clamp the result to between the 32-bit.
339 result = std::clamp(result, raw_min(), raw_max());
340
341 return percent(static_cast<int_t>(result));
342 }

451 [[nodiscard]] auto to_string() const noexcept
452 {
453 constexpr int_t fixed_percent_scalar = 1000000000;
454 // Based on the number of characters needed to hold "fixed_percent_scalar"
455 // as well as a '.' and a '-' sign
456 constexpr size_t scalar_length = 12;
457
458 if (raw_value() >= raw_max() - 2) {
459 return to_array<scalar_length>("+1.000000000");
460 } else if (raw_value() <= raw_min() + 2) {
461 return to_array<scalar_length>("-1.000000000");
462 }
463
464 // Make a copy of the percent value
465 percent absolute_percent = *this;
466 // Check and save if the number is negative
467 bool negative = absolute_percent.raw_value() < 0;
468 // If negative absolute, make the percent positive
469 if (negative) {
470 absolute_percent.m_value = absolute_percent.raw_value() * -1;
471 }
472
473 // +1 for a '\0' at the end
474 std::array<char, scalar_length + 1> buffer{ '\0' };
475 // +2 for a '.' and a '\0' character at the end
476 std::array<char, scalar_length + 1> percent_string{ '\0' };
477
478 // Scale "fixed_percent_scalar" by the absolute_percent value
479 int_t decimal_percent = absolute_percent * fixed_percent_scalar;
480
481 // Convert to characters
482 auto char_conversion_result =
483 std::to_chars(buffer.begin(), buffer.end(), decimal_percent);
484
485 size_t string_length = char_conversion_result.ptr - buffer.data();
486 size_t leading_zeros = scalar_length - string_length;
487
488 // Add the leading +/- sign
489 percent_string[0] = (negative) ? '-' : '+';
490
491 // Add the '0' and '.'. We know that the value must be below 1.0 and above
492 // -1.0 because if the values were 1.0 or -1.0 then we would have returned
493 // them at the start of the function.
494 percent_string[1] = '0';
495 percent_string[2] = '.';
496
497 // Add any leading zeros into the percent string array
498 for (size_t i = 3; i < leading_zeros; i++) {
499 percent_string[i] = '0';
500 }
501 // Now copy the contents of the character buffer into the percent string
502 // offset by the number of leading zeros.
503 std::copy_n(
504 buffer.begin(), string_length, percent_string.begin() + leading_zeros);
505
